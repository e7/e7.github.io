# -------- ssh --------
# ssh隧道（socks5协议）
ssh -f -N -D 127.0.0.1:1080 user@host
# 端口映射
ssh -N -L 3306:192.168.1.100:3306 -L 3307:192.168.1.101:3307 user@host

# base64的md5
echo -n foo | openssl dgst -md5 -binary | openssl enc -base64

# -------- git --------
# 基于版本创建分支
git checkout -b branch1 5ec8848d691874df46c11f4b624952f4a44397f2
# 漂亮的git-log
git log --graph --pretty=oneline --abbrev-commit
# git提交检出均不转换换行符
git config --global core.autocrlf false
# git拒绝提交包含混合换行符的文件
git config --global core.safecrlf true
# git存储凭证
git config --global credential.helper 'store --file /tmp/.git-credentails'
# 分支修剪，-n是dry-run
git remote prune origin -n
# 删除本地tag
git tag -d tag_name
# 删除远程tag
git push origin :refs/tags/tag_name
# 批量删除本地tag
git --no-pager tag -l | egrep 'v0\.0\.[1]?[0-9]$' | xargs git tag -d
# 批量删除远程tag
git show-ref --tag |  awk '/v0\.0\.[1]?[0-9]$/ {print ":"$2}' | xargs git push origin

# ubuntu导入gpg密钥
sudo apt-key adv --keyserver keyserver.ubuntu.com --recv-keys <PUBKEY>

# 非root用户使用低端口
sudo setcap CAP_NET_BIND_SERVICE=+eip /path/to/binary

# 隔离启动进程，模仿docker
unshare --fork --pid --mount-proc bash # 利用了clone系统调用创建namespace

# -------- grep --------
alias rgrep="rg --no-heading" # 类似grep的输出而不分组

# -------- ps --------
ps H -e0 pid,tid,%cpu | grep ${pid} # 查看某进程中cpu占用比较高的线程id

# -------- java --------
jstack ${pid} | grep ${tid} -A 30 # 查看某java线程的调用栈，注意tid需要16进制

# -------- docker --------
# 交互模式重启已经停止的容器（常用于恢复shell）
docker start -i 1234
# 获取容器ip地址
docker inspect -f '{{range .NetworkSettings.Networks}}{{.IPAddress}}{{end}}' container_name_or_id

# -------- ffmpeg --------
# 屏幕录像
ffmpeg -f x11grab -s xga -r 100  -i :0.0+0+0 -qscale 0.01 wheer.avi
