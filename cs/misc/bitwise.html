    # 位运算的骚操作

    • 第n位取反

      x ^ (1 << (n-1)); // 跟1异或就会取反

    • 取末n位

      x & (1 << (n-1));

    • 末n位变成1

      x | (1<<n - 1);

    • 末n位取反

      x ^ (1<<n - 1);

    • 末尾连续的1变为0

      x & (x+1);

    • 末尾连续的0变为1

      x | (x-1);

    • 第一个遇到的0变为1

      x | (x+1);

    • 取末尾连续的1

      (x ^ (x+1)) >> 1;

    • 取组成一个无符号整数的最小2次幂

      // 一个无符号整数可看成是由多个2次幂相加而成，相当于去掉遇到的第一个1的左侧
x & (x ^ (x-1))

    • 取32位整数的绝对值

      // 1. 因为补码的原因，计算机中取相反数就是按位取反加1
// 2. 于是问题就可以转化为，如果是负数就取反加1，正数就本身加0
// 3. 异或的性质为同0异1，所以x^0xFFFF则相当于取反，x^0x0000则相当与本身
x ^ (~(x>>31) + 1) + (x>>31);

    • 判断一个32位无符号整数中1个数的奇偶性

      int is_odd(uint32_t x) {
    x ^= x >> 1;
    x ^= x >> 2;
    x ^= x >> 4;
    x ^= x >> 8;
    x ^= x >> 16;
    return x&1;
}

    • 计算一个32位无符号整数中1的个数

      int count(uint32_t x) {
    x = (x & 0x55555555) + ((x >> 1) & 0x55555555);
	x = (x & 0x33333333) + ((x >> 2) & 0x33333333);
	x = (x & 0x0F0F0F0F) + ((x >> 4) & 0x0F0F0F0F);
	x = (x & 0x00FF00FF) + ((x >> 8) & 0x00FF00FF);
	return (x & 0x0000FFFF) + ((x >> 16) & 0x0000FFFF);
}
